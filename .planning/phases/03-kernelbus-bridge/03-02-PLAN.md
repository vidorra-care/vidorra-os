    **KernelBusHost implementation** (`packages/kernel/src/kernel-bus-host.ts`):

    ```typescript
    import type { KernelBusMessage, KernelBusResponse, KernelBusPush } from '@vidorra/types'
    import { themeEngine } from './theme-engine'
    import { useWindowStore } from '@vidorra/shell/stores/useWindowStore'  // adjust import path

    export class KernelBusHost {
      // Map from trusted contentWindow → windowId
      private trustedFrames = new Map<WindowProxy, string>()
      private unsubscribeTheme: (() => void) | null = null

      init(): void {
        window.addEventListener('message', this.handleMessage)
        this.unsubscribeTheme = themeEngine.subscribe((mode) => {
          const resolved = themeEngine.getResolvedMode()
          const push: KernelBusPush = {
            type: 'push',
            method: 'theme.changed',
            params: { mode: resolved },
          }
          for (const [frame] of this.trustedFrames) {
            frame.postMessage(push, '*')
          }
        })
      }

      destroy(): void {
        window.removeEventListener('message', this.handleMessage)
        this.unsubscribeTheme?.()
        this.trustedFrames.clear()
      }

      registerFrame(windowId: string, contentWindow: WindowProxy): void {
        // Do NOT add to trustedFrames yet — trust is granted only after app.ready
        // Store a pending registration so app.ready can look up the windowId
        this.pendingFrames.set(contentWindow, windowId)
      }

      unregisterFrame(windowId: string): void {
        for (const [frame, id] of this.trustedFrames) {
          if (id === windowId) { this.trustedFrames.delete(frame); break }
        }
        for (const [frame, id] of this.pendingFrames) {
          if (id === windowId) { this.pendingFrames.delete(frame); break }
        }
      }

      private pendingFrames = new Map<WindowProxy, string>()

      private handleMessage = (event: MessageEvent): void => {
        const data = event.data as KernelBusMessage & { method?: string }
        if (!data || typeof data !== 'object') return

        const source = event.source as WindowProxy | null
        if (!source) return

        // app.ready: one-way signal to promote pending → trusted
        if (data.method === 'app.ready') {
          const windowId = this.pendingFrames.get(source)
          if (windowId !== undefined) {
            this.trustedFrames.set(source, windowId)
            this.pendingFrames.delete(source)
          }
          return  // no response (D-12)
        }

        // All other methods: must be trusted
        if (!this.trustedFrames.has(source)) return  // silently drop (D-10)

        const { requestId, method, params } = data
        if (!requestId) return  // malformed message

        const windowId = this.trustedFrames.get(source)!
        this.dispatch(source, requestId, method, params, windowId)
      }

      private dispatch(
        source: WindowProxy,
        requestId: string,
        method: string,
        params: unknown,
        windowId: string
      ): void {
        const store = useWindowStore.getState()
        try {
          switch (method) {
            case 'window.setTitle': {
              const { title } = params as { title: string }
              useWindowStore.setState((s) => ({
                windows: s.windows.map((w) => w.id === windowId ? { ...w, title } : w),
              }))
              this.respond(source, { requestId })
              break
            }
            case 'window.close':
              store.closeWindow(windowId)
              this.respond(source, { requestId })
              break
            case 'window.minimize':
              store.setWindowState(windowId, 'minimized')
              this.respond(source, { requestId })
              break
            case 'window.maximize':
              store.toggleMaximize(windowId)
              this.respond(source, { requestId })
              break
            case 'window.resize': {
              const { width, height } = params as { width: number; height: number }
              const win = store.windows.find((w) => w.id === windowId)
              const currentRect = win?.rect ?? { x: 0, y: 0, width: 400, height: 300 }
              store.setWindowRect(windowId, { ...currentRect, width, height })
              this.respond(source, { requestId })
              break
            }
            case 'theme.get': {
              const mode = themeEngine.getResolvedMode()
              this.respond(source, { requestId, result: mode })
              break
            }
            default:
              this.respond(source, { requestId, error: `Unknown method: ${method}` })
          }
        } catch (err) {
          const message = err instanceof Error ? err.message : String(err)
          this.respond(source, { requestId, error: message })
        }
      }

      private respond(source: WindowProxy, response: KernelBusResponse): void {
        source.postMessage(response, '*')
      }
    }

    export const kernelBusHost = new KernelBusHost()
    ```

    **Import path for useWindowStore from kernel package:**
    The kernel package cannot import from shell (circular dependency). Instead, use dependency injection:
    - `KernelBusHost` should NOT import `useWindowStore` directly
    - Instead, accept store callbacks via `init(callbacks: KernelBusHostCallbacks)` where the shell passes them in
    - Define `KernelBusHostCallbacks` interface within kernel-bus-host.ts:

    ```typescript
    export interface KernelBusHostCallbacks {
      setWindowTitle: (windowId: string, title: string) => void
      closeWindow: (windowId: string) => void
      setWindowMinimized: (windowId: string) => void
      toggleMaximize: (windowId: string) => void
      setWindowRect: (windowId: string, rect: { x: number; y: number; width: number; height: number }) => void
      getWindowRect: (windowId: string) => { x: number; y: number; width: number; height: number } | undefined
      getResolvedTheme: () => 'light' | 'dark'
    }
    ```

    This is the correct architectural pattern: kernel is framework-agnostic, shell provides callbacks. The `init(callbacks)` signature replaces the direct store import.
